CREATE TABLE Clientes
(
	ID int AUTO_INCREMENT PRIMARY KEY,
	nombre VARCHAR(100),
	apellido VARCHAR(100),
	marca VARCHAR(100),
	modelo VARCHAR(100),
	matricula VARCHAR(100),
	autorizacion BOOLEAN,
	telefono INT
);
